Hexa to Decimal

a=list(map(str,input()))
    a.reverse()
    print(a)
    d=0
    for i in range(len(a)):
        if a[i]=='A':
            d = d + 10*(8**i)
        elif a[i]=='B':
            d = d + 11*(8**i)
        elif a[i]=='C':
            d = d + 12*(8**i)
        elif a[i]=='D':
            d = d + 13*(8**i)
        elif a[i]=='E':
            d = d + 14*(8**i)
        elif a[i]=='F':
            d = d + 15*(8**i)
        else:
            d = d + int(a[i])*(8**i)

    print(d)

Binary Search

    from array import *


def BinarySearch(nums,s):
    low = 0
    high = len(nums) - 1

    while(high >= low):
        mid = (low + high) // 2

        if nums[mid] == s:
            globals()['pos'] = mid
            return True
        else:
            if s > nums[mid]:
                low = mid + 1
            else:
                high = mid - 1

    return False

Bubble Sort

    def sort(nums):
    for i in range(len(nums)-1,0,-1):
        for j in range(i):
            if nums[j] > nums[j+1]:
                temp = nums[j]
                nums[j] = nums[j+1]
                nums[j+1] = temp




nums = [5, 6, 3, 8 ,7 ,2]

Selection Sort


def Sort(nums):

    for i in range(len(nums)):
        minpos = i

        for j in range(i,len(nums)):
            if nums[minpos] > nums[j]:
                minpos = j

        temp = nums[minpos]
        nums[minpos] = nums[i]
        nums[i] = temp

nums = [9, 3, 5, 7, 8 ,2]

Sort(nums)

print(nums)

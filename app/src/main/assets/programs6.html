    temp = num

    i = len(str(num))

        temp = temp//10

    if num==sum:
        print("Armstrong number")
    else:
        print("Not an armstrong number")

LCM and HCM

    m=arr[0]
    n=arr[1]

    if m>n:
        max=m
    else:
        max=n

    while(1):
        if (max%m==0 and max%n==0):
            print("{} is LCM".format(max))
            break
        max=max+1


    hcm=int((m*n)/max)

Perfect Number

    def perfect_number(n):
        sum = 0
        for x in range(1, n):
            if n % x == 0:
                sum += x
        return sum == n
    print(perfect_number(6))

    def fact(n):

        return n * fact(n-1)

    night = 0

        time[0] = 0
